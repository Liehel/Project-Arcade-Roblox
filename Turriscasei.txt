local function highlightButton(button)
	if not button then return end
	local underline = button:FindFirstChild("Underline")
	local originalSize = button:GetAttribute("OriginalSize")
	local originalColor = button:GetAttribute("OriginalColor")
	if not originalSize then return end
	local hoverSize = UDim2.new(originalSize.X.Scale * 1.1, originalSize.X.Offset, originalSize.Y.Scale * 1.1, originalSize.Y.Offset)
	local targetColor = button.BackgroundColor3
	if originalColor then targetColor = originalColor:Lerp(Color3.new(1, 1, 1), 0.3) end
	TweenService:Create(button, TweenInfo.new(0.2), { Size = hoverSize, BackgroundColor3 = targetColor }):Play()
	if underline then TweenService:Create(underline, TweenInfo.new(0.3), { Size = UDim2.new(1, 0, 0, 3) }):Play() end
end

local function unhighlightButton(button)
	if not button then return end
	local underline = button:FindFirstChild("Underline")
	local originalSize = button:GetAttribute("OriginalSize")
	local originalColor = button:GetAttribute("OriginalColor")
	if not originalSize then return end
	local targetColor = originalColor or button.BackgroundColor3
	TweenService:Create(button, TweenInfo.new(0.2), { Size = originalSize, BackgroundColor3 = targetColor }):Play()
	if underline then TweenService:Create(underline, TweenInfo.new(0.2), { Size = UDim2.new(0, 0, 0, 3) }):Play() end
end

local function updateSelectionVisuals()
	for i, btn in ipairs(currentButtons) do
		if i == selectedButtonIndex then highlightButton(btn) else unhighlightButton(btn) end
	end
end

local function setupStyledButtonBehavior(button)
	button:SetAttribute("OriginalSize", button.Size)
	button:SetAttribute("OriginalColor", button.BackgroundColor3)
	local underline = Instance.new("Frame", button)
	underline.Name = "Underline"
	underline.Size = UDim2.new(0, 0, 0, 3)
	underline.Position = UDim2.new(0.5, 0, 1, 0)
	underline.AnchorPoint = Vector2.new(0.5, 1)
	underline.BackgroundColor3 = COLOR_UNDERLINE
	underline.BorderSizePixel = 0
	underline.ZIndex = button.ZIndex + 3
	button.MouseEnter:Connect(function()
		for i, btn in ipairs(currentButtons) do
			if btn == button then
				if selectedButtonIndex ~= i then playSound(hoverSound) end
				selectedButtonIndex = i
				updateSelectionVisuals()
				break
			end
		end
	end)
end

local function createStyledButton(parent, text, position, size, colorHex, iconType)
	local button = Instance.new("TextButton", parent)
	button.Position = position
	button.Size = size
	if typeof(colorHex) == "Color3" then
		button.BackgroundColor3 = colorHex
	else
		button.BackgroundColor3 = colorHex and Color3.fromHex(colorHex) or COLOR_BTN_BASE
	end
	button.Text = ""
	button.BorderSizePixel = 0
	button.AnchorPoint = Vector2.new(0.5, 0.5)
	button.AutoButtonColor = false
	button:SetAttribute("ActionName", text)

	local corner = Instance.new("UICorner", button)
	corner.CornerRadius = UDim.new(0, 10)
	addGlassEffectToFrame(button)
	local content = Instance.new("Frame", button)
	content.Size = UDim2.new(1, 0, 1, 0)
	content.BackgroundTransparency = 1
	content.ZIndex = button.ZIndex + 2
	local listLayout = Instance.new("UIListLayout", content)
	listLayout.FillDirection = Enum.FillDirection.Horizontal
	listLayout.VerticalAlignment = Enum.VerticalAlignment.Center
	listLayout.HorizontalAlignment = Enum.HorizontalAlignment.Center
	listLayout.Padding = UDim.new(0, 10)

	if iconType then 
		local icon = createIcon(content, iconType, 30, COLOR_LABEL_BUTTON) 
		icon.LayoutOrder = 1 
	end
	local label = Instance.new("TextLabel", content)
	label.Size = UDim2.new(iconType and 0.6 or 0.9, 0, 1, 0)
	label.BackgroundTransparency = 1
	label.Text = text
	label.TextScaled = true
	label.TextColor3 = COLOR_LABEL_BUTTON
	label.Font = Enum.Font.GothamBold
	label.TextXAlignment = Enum.TextXAlignment.Center
	label.LayoutOrder = 2
	setupStyledButtonBehavior(button)
	return button
end

local function GetYellowGradientColor(score)
	local hue = 0.14 + (math.sin(score * 0.1) * 0.04) 
	return Color3.fromHSV(hue, 0.8, 1)
end

local function TriggerScreenShake(intensity)
	SHAKE_INTENSITY = intensity
	task.spawn(function()
		local startTime = tick()
		while (tick() - startTime) < 0.25 do
			if not inGame then break end
			local shake = Vector3.new(
				(math.random() - 0.5) * SHAKE_INTENSITY,
				(math.random() - 0.5) * SHAKE_INTENSITY,
				(math.random() - 0.5) * SHAKE_INTENSITY
			)
			local cam = workspace.CurrentCamera
			cam.CFrame = cam.CFrame * CFrame.new(shake)
			RunService.RenderStepped:Wait()
			cam.CFrame = cam.CFrame * CFrame.new(-shake)
		end
		SHAKE_INTENSITY = 0
	end)
end

end

-- 🔥 AQUI ESTÁ LA REPARACIÓN DE LA CÁMARA 🔥
local function Cleanup3DGame()
	if gameData.moveConnection then gameData.moveConnection:Disconnect() end
	if currentTowerFolder then currentTowerFolder:Destroy() end

	if hudGui then hudGui:Destroy(); hudGui = nil end
	if versusGui then versusGui:Destroy(); versusGui = nil end
	if waitingOverlay then waitingOverlay:Destroy(); waitingOverlay = nil end

	-- FORZAR EL REGRESO DE LA CÁMARA AL JUGADOR
	local cam = workspace.CurrentCamera
	cam.CameraType = Enum.CameraType.Custom
	cam.FieldOfView = 70
	if player.Character then
		local humanoid = player.Character:FindFirstChild("Humanoid")
		local hrp = player.Character:FindFirstChild("HumanoidRootPart")
		if humanoid and hrp then
			cam.CameraSubject = humanoid
			-- Teletransportar la cámara físicamente para que no se quede atrapada arriba
			cam.CFrame = CFrame.lookAt(hrp.Position + Vector3.new(0, 5, 12), hrp.Position)
		end
	end

	-- Restaurar la iluminación al mundo normal
	Lighting.TimeOfDay = originalLighting.TimeOfDay
	Lighting.Brightness = originalLighting.Brightness
	Lighting.Ambient = originalLighting.Ambient
	Lighting.OutdoorAmbient = originalLighting.OutdoorAmbient
	Lighting.ColorShift_Bottom = originalLighting.ColorShift_Bottom
	Lighting.ColorShift_Top = originalLighting.ColorShift_Top
	Lighting.FogColor = originalLighting.FogColor
	Lighting.FogEnd = originalLighting.FogEnd
	Lighting.FogStart = originalLighting.FogStart

	local customAtmo = Lighting:FindFirstChild("SunsetAtmosphere")
	if customAtmo then customAtmo:Destroy() end
	local customSky = Lighting:FindFirstChild("SunsetSky")
	if customSky then customSky:Destroy() end
	local customClouds = workspace.Terrain:FindFirstChild("SunsetClouds")
	if customClouds then customClouds:Destroy() end
end

local function HandleButtonAction(btn)
	if not btn then return end
	local action = btn:GetAttribute("ActionName")

	playSound(selectSound)

	if action == "PLAY" then
		if playerGui:FindFirstChild("TurrisArcadeMenu") then playerGui.TurrisArcadeMenu:Destroy() end
		createPlayModeMenu()

	elseif action == "VIP" or action == "TURRIS" then
		if playerGui:FindFirstChild("TurrisArcadeMenu") then playerGui.TurrisArcadeMenu:Destroy() end
		createDonateMenu()

	elseif action == "EXIT" then
		FullQuitGame()

	elseif action == "SOLO" then
		if playerGui:FindFirstChild("TurrisPlayModeMenu") then playerGui.TurrisPlayModeMenu:Destroy() end
		currentMenuType = nil
		table.clear(currentButtons)
		table.clear(buttonActions)
		inGame = true
		isVersusMode = false
		disablePlayerMovement()
		if StartRequest then pcall(function() StartRequest:FireServer("Solo") end) end
		Setup3DGame()

	elseif action == "VERSUS" then
		if playerGui:FindFirstChild("TurrisPlayModeMenu") then playerGui.TurrisPlayModeMenu:Destroy() end
		currentMenuType = nil 
		table.clear(currentButtons)
		table.clear(buttonActions)
		local success, LobbyClient = pcall(function()
			return require(player.PlayerScripts:WaitForChild("TurrisCaseiLobbyClient"))
		end)
		if success and LobbyClient and LobbyClient.OpenMenu then
			LobbyClient.OpenMenu()
		else
			warn("No se pudo cargar TurrisCaseiLobbyClient")
		end

	elseif action == "BACK" or action == "MENU" then
		ReturnToMainMenu()

	elseif action == "EXIT_GAME" then
		FullQuitGame()

	elseif string.find(action, "DONATE") then
		local id = tonumber(string.match(action, "%d+"))
		if id then MarketplaceService:PromptProductPurchase(player, id) end
	end
end

local function ScrollTowerDown()
	local THRESHOLD = 6
	local bHeight = (TurrisModule.Settings and TurrisModule.Settings.BlockHeight) or 1.5
	if gameData.blocksCount > THRESHOLD then
		for _, obj in ipairs(currentTowerFolder:GetChildren()) do
			if obj.Name ~= "RoomModel" then
				local newY = obj.Position.Y - bHeight
				TweenService:Create(obj, TweenInfo.new(0.3, Enum.EasingStyle.Quad), {
					Position = Vector3.new(obj.Position.X, newY, obj.Position.Z)
				}):Play()
				if newY < (GAME_ORIGIN.Y - (bHeight * 1)) then obj:Destroy() end
			end
		end
	end
end

local function SpawnNextBlock()

	local limit = gameData.limit or 15

	if gameData.direction == "X" then
		spawnPos = prevPos + Vector3.new(-limit, bHeight, 0)
	else
		spawnPos = prevPos + Vector3.new(0, bHeight, -limit)
	end

	local nextBlockData = TurrisModule.PickNextBlockData()
	gameData.activeBlockData = nextBlockData

	local finalColor = nextBlockData.Color
	if nextBlockData.Type == "Normal" then
		finalColor = GetYellowGradientColor(gameData.score)

	newBlock.CFrame = CFrame.new(spawnPos)

	alignLight.Angle = 90
	alignLight.Brightness = 0

		sg.Face = Enum.NormalId.Top
		sg.CanvasSize = Vector2.new(100, 100)

local function StartMovementLoop()
	if gameData.moveConnection then gameData.moveConnection:Disconnect() end

	gameData.phase = -math.pi / 2 

	gameData.moveConnection = RunService.RenderStepped:Connect(function(dt)
		if not inGame or not gameData.currentBlock then return end

		if isVersusMode then
			local elapsed = tick() - turnTimer
			if gameData.score == 0 and elapsed > MAX_INITIAL_WAIT then 
				inGame = false 
				if PlayerDiedEvent then PlayerDiedEvent:FireServer() end
				return 
			end
			if gameData.score > 0 and elapsed > MAX_TURN_TIME then
				turnPenalty = (elapsed - MAX_TURN_TIME) * 5
			else
				turnPenalty = 0
			end
		end

		local sx = gameData.previousBlock.Size.X
		local sz = gameData.previousBlock.Size.Z
		local currentSize = (gameData.direction == "X") and sx or sz

		local sizeFactor = math.clamp(currentSize / 10, 0.4, 1.0) 
		local currentSpeed = ((gameData.speed or 20) * sizeFactor) + turnPenalty

		local MAX_ALLOWED_SPEED = 30 
		if currentSpeed > MAX_ALLOWED_SPEED then currentSpeed = MAX_ALLOWED_SPEED end

		gameData.phase = gameData.phase + (dt * (currentSpeed / 10))

		local limit = gameData.limit or 15
		local offset = math.sin(gameData.phase) * limit

		local basePos = gameData.startPos
		local targetPos

		if gameData.direction == "X" then
			targetPos = Vector3.new(gameData.previousBlock.Position.X + offset, basePos.Y, basePos.Z)
		else
			targetPos = Vector3.new(basePos.X, basePos.Y, gameData.previousBlock.Position.Z + offset)
		end

		gameData.currentBlock.Position = targetPos

		local diff = 0
		if gameData.direction == "X" then
			diff = targetPos.X - gameData.previousBlock.Position.X
		else
			diff = targetPos.Z - gameData.previousBlock.Position.Z
		end

		local tolerance = (TurrisModule and TurrisModule.Settings and TurrisModule.Settings.Tolerance) or 3.0
		local light = gameData.currentBlock:FindFirstChild("AlignLight")
		if light then
			if math.abs(diff) <= tolerance then
				light.Brightness = 10 
				gameData.currentBlock.Material = Enum.Material.Neon
			else
				light.Brightness = 0
				if gameData.activeBlockData and gameData.activeBlockData.Type == "Normal" then
					gameData.currentBlock.Material = Enum.Material.SmoothPlastic
				end
			end
		end
	end)
end

local function UpdateDynamicCamera()
	local cam = workspace.CurrentCamera
	if inGame and gameData.currentBlock then
		local sx = gameData.currentBlock.Size.X
		local sz = gameData.currentBlock.Size.Z
		local minDim = math.min(sx, sz)

		local targetFOV = 70
		local targetPos = GAME_ORIGIN + Vector3.new(13, 28, 15) 

		local lookAtPos = Vector3.new(GAME_ORIGIN.X, gameData.currentBlock.Position.Y - 2, GAME_ORIGIN.Z)

		if minDim < 6.0 then
			local percent = math.clamp(minDim / 6.0, 0.2, 1.0)

			targetFOV = 50 + (20 * percent) 

			local offsetX = 8 + (5 * percent)
			local offsetY = 22 + (6 * percent) 
			local offsetZ = 10 + (5 * percent)

			targetPos = GAME_ORIGIN + Vector3.new(offsetX, offsetY, offsetZ)
		end

		local tweenInfo = TweenInfo.new(0.8, Enum.EasingStyle.Quad, Enum.EasingDirection.Out)
		TweenService:Create(cam, tweenInfo, {
			FieldOfView = targetFOV,
			CFrame = CFrame.lookAt(targetPos, lookAtPos)
		}):Play()
	end
end

local function PlaceBlock()
	if not inGame or not gameData.currentBlock then return end
	gameData.moveConnection:Disconnect()

	local success, newSize, offset, fallingPartData, bonusScore = TurrisModule.CalculateStack(
		gameData.previousBlock.Size, gameData.previousBlock.CFrame, gameData.currentBlock.CFrame, gameData.activeBlockData
	)

	if not success then
		gameData.currentBlock.Anchored = false
		Debris:AddItem(gameData.currentBlock, 2)
		playSound(gameOverSound)
		TriggerScreenShake(0.8)
		inGame = false 

		if isVersusMode then
			if PlayerDiedEvent then PlayerDiedEvent:FireServer() end

			showWaitingOverlay("YOU DIED!", "Waiting for winner...", Color3.fromRGB(255, 100, 100))
			setBlurEnabled(true, 56)
		else
			task.wait(1)
			showEndOverlay(gameData.score, false)
		end
		return
	end

	local bHeight = TurrisModule.Settings.BlockHeight or 1.5
	local prevPos = gameData.previousBlock.Position
	local newCenter = Vector3.new(prevPos.X + offset.X, prevPos.Y + bHeight, prevPos.Z + offset.Z)

	gameData.currentBlock.Size = newSize
	gameData.currentBlock.CFrame = CFrame.new(newCenter)

	for _, v in ipairs(gameData.currentBlock:GetChildren()) do
		if v:IsA("SurfaceGui") or v:IsA("SpotLight") then
			v:Destroy()
		end
	end

	PopBlock(gameData.currentBlock)
	FlashBlock(gameData.currentBlock)

	if fallingPartData then
		COMBO_COUNTER = 0
		playSound(placeSound)
		SpawnCutParticles(newCenter, gameData.currentBlock.Color)
		TriggerScreenShake(0.3)

		local waste = gameData.currentBlock:Clone()
		waste.Name = "Waste"; waste.Anchored = false;
		waste.CanCollide = false
		waste:FindFirstChild("Highlight"):Destroy()
		for _, v in ipairs(waste:GetChildren()) do if v:IsA("SurfaceGui") then v:Destroy() end end

		Debris:AddItem(waste, 0.3)

		if gameData.direction == "X" then
			waste.Size = Vector3.new(math.abs(fallingPartData.diffX), newSize.Y, newSize.Z)
			local wasteX = (fallingPartData.diffX > 0) and (newCenter.X + newSize.X/2 + waste.Size.X/2) or (newCenter.X - newSize.X/2 - waste.Size.X/2)
			waste.Position = Vector3.new(wasteX, newCenter.Y, newCenter.Z)
		else
			waste.Size = Vector3.new(newSize.X, newSize.Y, math.abs(fallingPartData.diffZ))
			local wasteZ = (fallingPartData.diffZ > 0) and (newCenter.Z + newSize.Z/2 + waste.Size.Z/2) or (newCenter.Z - newSize.Z/2 - waste.Size.Z/2)
			waste.Position = Vector3.new(newCenter.X, newCenter.Y, wasteZ)
		end
		waste.Parent = currentTowerFolder
	else
		COMBO_COUNTER = COMBO_COUNTER + 1
		playSound(perfectSound)
		SpawnShockwave(newCenter)
		TriggerScreenShake(0.1)
		if COMBO_COUNTER > 1 then
			gameData.currentBlock.Material = Enum.Material.Neon
			if gameData.activeBlockData.Type == "Normal" then
				gameData.currentBlock.Color = Color3.fromHSV(0.16, 1, 1)
			end
		end
	end

	local points = 1
	if type(bonusScore) == "number" then
		points = 1 + bonusScore
		if bonusScore > 0 then
			ShowBonusText(newCenter, "+"..tostring(bonusScore).."!", Color3.new(0,1,0))
		elseif bonusScore < 0 then
			ShowBonusText(newCenter, tostring(bonusScore), Color3.new(1,0,0))
		end
	end

	gameData.score = gameData.score + points
	gameData.blocksCount = gameData.blocksCount + 1
	gameData.previousBlock = gameData.currentBlock

	gameData.speed = TurrisModule.GetNextSpeed(gameData.speed)

	UpdateDynamicCamera()
	if isVersusMode and UpdateLiveScore then UpdateLiveScore:FireServer(gameData.score) end
	if hudScoreLabel then hudScoreLabel.Text = "SCORE: " .. tostring(gameData.score) end
	ScrollTowerDown()
	SpawnNextBlock()
	StartMovementLoop()
end

UserInputService.InputBegan:Connect(function(input, processed)
	if input.KeyCode == Enum.KeyCode.Escape then
		if currentMenuType == "Main" then
			FullQuitGame()
		elseif currentMenuType == "PlayMode" or currentMenuType == "Donate" then
			ReturnToMainMenu()
		elseif currentMenuType == "GameOver" then
			FullQuitGame()
		elseif inGame then
			if isVersusMode then
				local lobbyRemotes = ReplicatedStorage.ArcadeRemotes.TurrisCasei.LobbySystem
				if lobbyRemotes then lobbyRemotes.LeaveRoom:FireServer() end
				FullQuitGame()
			else
				FullQuitGame()
			end
		end
		return
	end

	if inGame then
		local isClickOrTouch = (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch)
		local isSpace = (input.KeyCode == Enum.KeyCode.Space)

		if isClickOrTouch or isSpace then
			if processed and isClickOrTouch then return end 
			PlaceBlock()
			return
		end
	end

	if processed then return end

	if currentMenuType and #currentButtons > 0 then
		if input.KeyCode == Enum.KeyCode.W or input.KeyCode == Enum.KeyCode.Up or input.KeyCode == Enum.KeyCode.A or input.KeyCode == Enum.KeyCode.Left then
			selectedButtonIndex = selectedButtonIndex - 1
			if selectedButtonIndex < 1 then selectedButtonIndex = #currentButtons end
			playSound(hoverSound);
			updateSelectionVisuals()

		elseif input.KeyCode == Enum.KeyCode.S or input.KeyCode == Enum.KeyCode.Down or input.KeyCode == Enum.KeyCode.D or input.KeyCode == Enum.KeyCode.Right then
			selectedButtonIndex = selectedButtonIndex + 1
			if selectedButtonIndex > #currentButtons then selectedButtonIndex = 1 end
			playSound(hoverSound);
			updateSelectionVisuals()

		elseif input.KeyCode == Enum.KeyCode.Return or input.KeyCode == Enum.KeyCode.Space or input.KeyCode == Enum.KeyCode.KeypadEnter then
			local btn = currentButtons[selectedButtonIndex]
			if btn then
				if buttonActions[btn] then
					buttonActions[btn]()
				else
					HandleButtonAction(btn)
				end
			end
		end
	end
end)

local function UpdateCamera()
	if not inGame or not gameData.dummy then return end
	local cam = workspace.CurrentCamera
	local targetPlatform = gameData.targetPlatform

	local lookPos = gameData.dummy.Position
	if targetPlatform then

local function SpawnNextPlatform()
	if not inGame then return end

	local pData = TapModule.PickNextPlatformData()
	local modelName = pData.ModelName

	local prevPlatform = gameData.platforms[#gameData.platforms]
	local prevPos = prevPlatform.Position

	local distance = math.random(TapModule.Settings.MinDistance, TapModule.Settings.MaxDistance)

	local axis = (math.random() > 0.5) and Vector3.new(1, 0, 0) or Vector3.new(0, 0, 1)

	local template = Assets:FindFirstChild(modelName)
	if not template then return end

	local newPlatform = template:Clone()
	newPlatform.Anchored = true 

	local newY = GLOBAL_TOP_Y - (newPlatform.Size.Y / 2)
	local targetXZ = prevPos + (axis * distance)
	local spawnPos = Vector3.new(targetXZ.X, newY, targetXZ.Z)

	newPlatform.CFrame = CFrame.new(spawnPos)
	newPlatform:SetAttribute("PlatformType", pData.Type)
	newPlatform:SetAttribute("PlatformValue", pData.Value)

	TapModule.ApplyRoseGradient(newPlatform, pData.Color)
	newPlatform.Parent = currentTowerFolder

	table.insert(gameData.platforms, newPlatform)
	gameData.targetPlatform = newPlatform

	if #gameData.platforms > TapModule.Settings.MaxVisiblePlatforms then
		local old = table.remove(gameData.platforms, 1)
		if old then old:Destroy() end
	end

	local startY = newPlatform.Position.Y
	newPlatform.Position = newPlatform.Position - Vector3.new(0, 10, 0)
	TweenService:Create(newPlatform, TweenInfo.new(0.5, Enum.EasingStyle.Back, Enum.EasingDirection.Out), {
		Position = Vector3.new(newPlatform.Position.X, startY, newPlatform.Position.Z)
	}):Play()

	local lookTarget = Vector3.new(spawnPos.X, gameData.dummy.Position.Y, spawnPos.Z)
	local playerTargetCFrame = CFrame.lookAt(gameData.dummy.Position, lookTarget)
	gameData.rotationTween = TweenService:Create(gameData.dummy, TweenInfo.new(0.25), {CFrame = playerTargetCFrame})
	gameData.rotationTween:Play()

	UpdateCamera()
end

local function checkLandingTolerant(finalPos, platform)
	local isInside, isPerfect = TapModule.CheckLanding(finalPos, platform)
	if not isInside then
		local pSize = platform.Size
		local pPos = platform.Position
		local diffX = math.abs(finalPos.X - pPos.X)
		local diffZ = math.abs(finalPos.Z - pPos.Z)
		if diffX < (pSize.X / 2 + 1.5) and diffZ < (pSize.Z / 2 + 1.5) then
			isInside = true
			isPerfect = false
		end
	end
	return isInside, isPerfect
end

local function HandleLanding(finalPos)
	local platform = gameData.targetPlatform
	local isInside, isPerfect = checkLandingTolerant(finalPos, platform)

	local trail = gameData.dummy:FindFirstChild("JumpTrail")
	if trail then trail.Enabled = false end

	if isInside then
		if isPerfect then
			playSound(perfectSound)
			gameData.score = gameData.score + 3
			ShowBonusText(finalPos, "PERFECT!\n+3", Color3.new(1,1,0))
			SpawnParticleBurst(finalPos, true)
		else
			playSound(jumpSound)
			gameData.score = gameData.score + 1
			SpawnParticleBurst(finalPos, false)
		end

		local type = platform:GetAttribute("PlatformType")
		local val = platform:GetAttribute("PlatformValue") or 0
		if type == "Bonus" then
			gameData.score = gameData.score + val
			ShowBonusText(finalPos + Vector3.new(0,2,0), "+"..val.." BONUS!", TapModule.Settings.Colors.Bonus)
		elseif type == "Penalty" then
			gameData.score = math.max(0, gameData.score + val)
			ShowBonusText(finalPos + Vector3.new(0,2,0), val.." PENALTY", TapModule.Settings.Colors.Penalty)
		end

		if hudScoreLabel then hudScoreLabel.Text = "SCORE: " .. tostring(gameData.score) end
		if isVersusMode and UpdateLiveScore then UpdateLiveScore:FireServer(gameData.score) end

		SpawnNextPlatform()
	else
		gameData.dummy.Anchored = false
		playSound(gameOverSound)

		task.wait(1.5)
		if isVersusMode then
			inGame = false 
			if MatchStatus then MatchStatus:FireServer("GameOver", gameData.score, 0) end
			if PlayerDiedEvent then PlayerDiedEvent:FireServer() end

			if waitingOverlay then waitingOverlay:Destroy() end
			local gui = Instance.new("ScreenGui", playerGui)
			gui.Name = "WaitingOverlay"
			local f = Instance.new("Frame", gui)
			f.Size = UDim2.fromOffset(400, 150)
			f.Position = UDim2.fromScale(0.5, 0.5)
			f.AnchorPoint = Vector2.new(0.5, 0.5)
			f.BackgroundColor3 = Color3.new(0,0,0)
			f.BackgroundTransparency = 0.5
			local lbl = Instance.new("TextLabel", f)
			lbl.Size = UDim2.fromScale(1, 1)
			lbl.BackgroundTransparency = 1
			lbl.Text = "YOU DIED!\nWaiting for winner..."
			lbl.Font = Enum.Font.GothamBlack
			lbl.TextColor3 = Color3.new(1,0,0)
			lbl.TextScaled = true
			waitingOverlay = gui
		else
			endGame()
		end
	end
end

local function StartCharge()
	if not inGame or gameData.isJumping or gameData.isCharging then return end

	if gameData.rotationTween then
		gameData.rotationTween:Cancel()
		if gameData.targetPlatform then
			local lookTarget = Vector3.new(gameData.targetPlatform.Position.X, gameData.dummy.Position.Y, gameData.targetPlatform.Position.Z)
			gameData.dummy.CFrame = CFrame.lookAt(gameData.dummy.Position, lookTarget)
		end
	end

	gameData.isCharging = true
	gameData.chargeStartTime = tick()

	if gameData.startPromptText then
		gameData.startPromptText:Destroy()
		gameData.startPromptText = nil
	end

	if chargeSound then chargeSound:Play() end

	local emitter = gameData.dummy:FindFirstChild("ChargeEmitter", true)
	if emitter then emitter.Enabled = true end

	if gameData.scaleTween then gameData.scaleTween:Cancel() end
	gameData.scaleVal = Instance.new("NumberValue")
	gameData.scaleVal.Value = 1 

	gameData.scaleConnection = gameData.scaleVal.Changed:Connect(function()
		if player.Character then
			player.Character:ScaleTo(gameData.scaleVal.Value)
		end
	end)

	gameData.scaleTween = TweenService:Create(gameData.scaleVal, TweenInfo.new(MAX_CHARGE_TIME, Enum.EasingStyle.Linear), {Value = 0.6})
	gameData.scaleTween:Play()
end

local function ReleaseCharge()
	if not inGame or not gameData.isCharging then return end
	gameData.isCharging = false
	gameData.isJumping = true

	if chargeSound then chargeSound:Stop() end

	local emitter = gameData.dummy:FindFirstChild("ChargeEmitter", true)
	if emitter then emitter.Enabled = false end

	if gameData.scaleTween then gameData.scaleTween:Cancel() end
	if gameData.scaleConnection then gameData.scaleConnection:Disconnect() end
	if gameData.scaleVal then gameData.scaleVal:Destroy() end
	if player.Character then player.Character:ScaleTo(1) end

	local holdTime = tick() - gameData.chargeStartTime
	holdTime = math.min(holdTime, MAX_CHARGE_TIME) 

	local vForward, vUp = TapModule.CalculateJumpVelocity(holdTime)
	local gravity = TapModule.Settings.Gravity

	local startPos = gameData.dummy.Position
	local timeInAir = 0

	local forwardDir = gameData.dummy.CFrame.LookVector

	local trail = gameData.dummy:FindFirstChild("JumpTrail")
	if trail then trail.Enabled = true end

	local legOffset = getCharacterLegOffset(gameData.humanoid, gameData.dummy)
	local perfectLandingY = GLOBAL_TOP_Y + legOffset

	if gameData.humanoid then
		gameData.humanoid:ChangeState(Enum.HumanoidStateType.PlatformStanding)
	end

	gameData.jumpConnection = RunService.RenderStepped:Connect(function(dt)
		timeInAir = timeInAir + dt

		local currentXZ = forwardDir * (vForward * timeInAir)
		local currentY = (vUp * timeInAir) - (0.5 * gravity * timeInAir^2)

		local newPos = startPos + currentXZ + Vector3.new(0, currentY, 0)
		gameData.dummy.CFrame = gameData.dummy.CFrame.Rotation + newPos

		UpdateCamera()

		if currentY < 0 and newPos.Y <= perfectLandingY then
			gameData.jumpConnection:Disconnect()
			gameData.jumpConnection = nil

			gameData.dummy.CFrame = gameData.dummy.CFrame.Rotation + Vector3.new(newPos.X, perfectLandingY, newPos.Z)
			gameData.isJumping = false

			if gameData.humanoid then
				gameData.humanoid:ChangeState(Enum.HumanoidStateType.Landed)
			end

			HandleLanding(gameData.dummy.Position)
		elseif newPos.Y < GAME_ORIGIN.Y - 50 then
			gameData.jumpConnection:Disconnect()
			gameData.isJumping = false
			HandleLanding(newPos) 
		end
	end)
end

local function Cleanup3DGame()

	if currentTowerFolder then currentTowerFolder:Destroy() end

	if hudGui then hudGui:Destroy() hudGui = nil end
	if versusGui then versusGui:Destroy() versusGui = nil end
	if waitingOverlay then waitingOverlay:Destroy() waitingOverlay = nil end

	if gameData.startPromptText then
		gameData.startPromptText:Destroy()
		gameData.startPromptText = nil
	end

	if not isVersusMode then
		local cam = workspace.CurrentCamera
		cam.CameraType = Enum.CameraType.Custom
		cam.FieldOfView = 70
	end

	if gameData.dummy then
		local emitter = gameData.dummy:FindFirstChild("ChargeEmitter", true)
		if emitter then emitter.Enabled = false end
	end

	local character = player.Character
	if character and gameData.originalCharacterCFrame then
		local hrp = character:FindFirstChild("HumanoidRootPart")
		if hrp then
			hrp.Anchored = false
			hrp.Velocity = Vector3.new(0, 0, 0)
		end
		character:ScaleTo(1)
		character:PivotTo(gameData.originalCharacterCFrame)
		gameData.originalCharacterCFrame = nil
	end

	Lighting.TimeOfDay = originalLighting.TimeOfDay
	Lighting.Brightness = originalLighting.Brightness
	Lighting.Ambient = originalLighting.Ambient
	Lighting.OutdoorAmbient = originalLighting.OutdoorAmbient
	Lighting.ColorShift_Bottom = originalLighting.ColorShift_Bottom

	Lighting.FogColor = originalLighting.FogColor
	Lighting.FogEnd = originalLighting.FogEnd
	Lighting.FogStart = originalLighting.FogStart

	if customAtmo then customAtmo:Destroy() end

local function HandleButtonAction(btn)
	if not btn then return end
	local action = btn:GetAttribute("ActionName")
	playSound(selectSound)

	if action == "PLAY" then
		if playerGui:FindFirstChild("TapArcadeMenu") then playerGui.TapArcadeMenu:Destroy() end
		createPlayModeMenu()
	elseif action == "VIP" or action == "TAP" then
		if playerGui:FindFirstChild("TapArcadeMenu") then playerGui.TapArcadeMenu:Destroy() end
		createDonateMenu()
	elseif action == "EXIT" then
		FullQuitGame()
	elseif action == "SOLO" then
		if playerGui:FindFirstChild("TapPlayModeMenu") then playerGui.TapPlayModeMenu:Destroy() end
		currentMenuType = nil
		table.clear(currentButtons)
		table.clear(buttonActions)
		inGame = true
		isVersusMode = false
		disablePlayerMovement()
		if StartRequest then pcall(function() StartRequest:FireServer("Solo") end) end
		Setup3DGame()
	elseif action == "VERSUS" then
		if playerGui:FindFirstChild("TapPlayModeMenu") then playerGui.TapPlayModeMenu:Destroy() end
		currentMenuType = nil 
		table.clear(currentButtons)
		table.clear(buttonActions)
		local success, LobbyClient = pcall(function()
			return require(player.PlayerScripts:WaitForChild("TapAndHoldLobbyClient"))
		end)
		if success and LobbyClient and LobbyClient.OpenMenu then
			LobbyClient.OpenMenu()
		end
	elseif action == "BACK" or action == "MENU" then
		ReturnToMainMenu()
	elseif string.find(action, "DONATE") then
		local id = tonumber(string.match(action, "%d+"))
		if id then MarketplaceService:PromptProductPurchase(player, id) end
	end
end

UserInputService.InputBegan:Connect(function(input, processed)
	if input.KeyCode == Enum.KeyCode.Escape then
		if currentMenuType == "Main" or currentMenuType == "GameOver" then
			FullQuitGame()
		elseif currentMenuType == "PlayMode" or currentMenuType == "Donate" then
			ReturnToMainMenu()
		elseif inGame then
			if isVersusMode then
				local lobbyRemotes = ReplicatedStorage.ArcadeRemotes.TapAndHold.LobbySystem
				if lobbyRemotes then lobbyRemotes.LeaveRoom:FireServer() end
			end
			FullQuitGame()
		end
		return
	end

	if inGame and not processed then
		if input.KeyCode == Enum.KeyCode.Space or input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			StartCharge()
		end
		return
	end

	if processed then return end

	if currentMenuType and #currentButtons > 0 then
		if input.KeyCode == Enum.KeyCode.W or input.KeyCode == Enum.KeyCode.Up or input.KeyCode == Enum.KeyCode.A or input.KeyCode == Enum.KeyCode.Left then
			selectedButtonIndex = selectedButtonIndex - 1
			if selectedButtonIndex < 1 then selectedButtonIndex = #currentButtons end
			playSound(hoverSound);
			updateSelectionVisuals()
		elseif input.KeyCode == Enum.KeyCode.S or input.KeyCode == Enum.KeyCode.Down or input.KeyCode == Enum.KeyCode.D or input.KeyCode == Enum.KeyCode.Right then
			selectedButtonIndex = selectedButtonIndex + 1
			if selectedButtonIndex > #currentButtons then selectedButtonIndex = 1 end
			playSound(hoverSound);
			updateSelectionVisuals()
		elseif input.KeyCode == Enum.KeyCode.Return or input.KeyCode == Enum.KeyCode.Space or input.KeyCode == Enum.KeyCode.KeypadEnter then
			local btn = currentButtons[selectedButtonIndex]
			if btn and buttonActions[btn] then
				buttonActions[btn]()
			end
		end
	end
end)

UserInputService.InputEnded:Connect(function(input, processed)
	if inGame and not processed then
		if input.KeyCode == Enum.KeyCode.Space or input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			ReleaseCharge()
		end
	end
end)
